76 #if defined(__AVR__)

80 #ifdef TLC_ATMEGA_8_H
81  uint8_t oldTCCR2 = TCCR2;

86 #elif defined(TLC_TIMER3_GSCLK)

92  TCNT2 = 0;
93  OCR2A = SERVO_TIMER2_TOP;
94  TCCR2B = oldTCCR2B;
95 #endif
96  TCCR1B |= _BV(CS11); // start timer1 with div 8 prescale
97 
98 #elif defined(__arm__) && defined(TEENSYDUINO)
99  //clear_XLAT_interrupt();
100  uint32_t sc = FTM1_SC;

133 uint16_t tlc_angleToVal(uint8_t angle)
134 {
135  return 4095 - SERVO_MIN_WIDTH - (
136  ((uint16_t)(angle) * (uint16_t)(SERVO_MAX_WIDTH - SERVO_MIN_WIDTH))
137  / SERVO_MAX_ANGLE);
138 }

142 uint8_t tlc_valToAngle(uint16_t value)
143 {
144  return SERVO_MAX_ANGLE * (4095 - SERVO_MIN_WIDTH - value)
145  / (SERVO_MAX_WIDTH - SERVO_MIN_WIDTH);
