Sets the grayscale data from an array in progmem. 

This doesn\textquotesingle{}t shift out any data\+: call Tlc.\+update(). An example\+: 
\begin{DoxyCode}
1 #include "tlc\_progmem\_utils.h"
2 prog\_uint8\_t gsArray1[NUM\_TLCS * 24] = \{
3   GS\_DUO((4095 * 16)/16, (4095 * 15)/16), GS\_DUO((4095 * 14)/16, (4095 * 13)/16),
4   GS\_DUO((4095 * 12)/16, (4095 * 11)/16), GS\_DUO((4095 * 10)/16, (4095 * 9)/16),
5   GS\_DUO((4095 * 8)/16, (4095 * 7)/16), GS\_DUO((4095 * 6)/16, (4095 * 5)/16),
6   GS\_DUO((4095 * 4)/16, (4095 * 3)/16), GS\_DUO((4095 * 2)/16, (4095 * 1)/16),
7 \};
8 
9 // sometime after Tlc.init()
10 tlc\_setGSfromProgmem(gsArray1);
11 Tlc.update();
\end{DoxyCode}
